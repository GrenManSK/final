36def download(url: str, fname: str, chunk_size: int = 1024) -> bool:
37 """
38 Download a file from a URL to a local file.
39
40 The first line is the function's signature. It's a single line of code that tells you everything
41 you need to know about the function
42
43 :param url: The URL of the file to download
44 :type url: str
45 :param fname: The name of the file to be downloaded
46 :type fname: str
47 :param chunk_size: The size of the chunks to download, defaults to 1024 (optional)
48 """
49 if os.path.exists(fname):
50 logging.warning(
51 'file %s already exists, still downloading and replacing', fname)
52 try:
53 resp = requests.get(url, stream=True)
54 total: int = int(resp.headers.get('content-length', 0))
55 with open(fname, 'wb') as file, tqdm(
56 desc=fname,
57 total=total,
58 unit='iB',
59 unit_scale=True,
60 unit_divisor=1024,
61 ) as bar:
62 for data in resp.iter_content(chunk_size=chunk_size):
63 size = file.write(data)
64 bar.update(size)
65 except ConnectionError:
66 logging.error("Failed to download file: %s", fname)
67 return False
68 return True
